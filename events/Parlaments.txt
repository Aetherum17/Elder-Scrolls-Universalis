#Rotten Borough
#Punishes awarding seat to poor and easily bribed provinces.
province_event = {
	id = parlaments.2
	
	title = parlaments.2.t
	desc = parlaments.2.d
	picture = PARLIAMENT_eventPicture

	trigger = {
		has_dlc = "Common Sense"
		owner = {
			has_parliament = yes
		}
		has_seat_in_parliament = yes
		is_capital = no
		NOT = { has_province_modifier = rotten_borough }
		NOT = { development = 10 }
		owner = {
			any_owned_province = {
				development = 10
				is_core = owner
				has_seat_in_parliament = no
				is_overseas = no
				has_owner_accepted_culture = yes
			}
		}
	}
	
	is_mtth_scaled_to_size = 1
	
	mean_time_to_happen = {
		months = 200
		modifier = {
			factor = 0.8
			NOT = { development = 6 }
		}
		modifier = {
			factor = 0.7
			NOT = { development = 4 }
		}
	}

	option = {
		name = parlaments.2.a
		ai_chance = {
			factor = 50
		}
		owner = { add_corruption = 0.5 }
		custom_tooltip = "parlaments.2.a.tt"
		add_province_modifier = {
			name = "rotten_borough"
			duration = -1
		}
	}
	option = {
		name = parlaments.2.b
		ai_chance = {
			factor = 50
			modifier = {
				factor = 4
				owner = {
					num_of_owned_provinces_with = {
						value = 2
						has_province_modifier = "rotten_borough"
					}
				}
			}
		}
		set_seat_in_parliament = no
		owner = {
			random_owned_province = {
				limit = {
					is_overseas = no
					is_core = owner
					has_owner_accepted_culture = yes
					development = 10
				}
				set_seat_in_parliament = yes
			}
			add_stability = -1
		}
		add_province_modifier = {
			name = "withdrawn_parliamentary_seat"
			duration = 3650
		}
	}
}
