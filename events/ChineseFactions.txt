country_event = {
	id = china_events.3
	title = "china_events.3.t"
	desc = "china_events.3.d"
	picture = COURT_eventPicture

	trigger = {
		OR = {
			faction_in_power = es_monks
			AND = {
				has_dlc = "Mandate of Heaven"
				is_emperor_of_china = yes
			}
		}
	}

		days = 1
	}

		name = "china_events.3.a"

		boost_eunuchs_effect = yes
		add_stability_or_adm_power = yes
	}

		name = "china_events.3.b"

			limit = { NOT = { has_dlc = "Mandate of Heaven" } }
			add_legitimacy = 10

		OR = {
			faction_in_power = es_monopolists

				meritocracy = 70
			}
		}
		NOT = { gold = 2 }
		any_owned_province = {
			NOT = { trade_goods = gold }
			has_port = no
			NOT = { development = 10 }
		}
	}

		days = 1
	}
	
	option = {
		name = "china_events.20.a"
		random_owned_province = {
			limit = {
				NOT = { trade_goods = gold }
				NOT = { development = 10 }
				has_port = no
			}
			change_trade_goods = gold
		}
		boost_bureaucrats_effect = yes
	}
	
	option = {
		name = "china_events.20.b"
		add_inflation = -5
		add_mandate_effect = yes
		reduce_bureaucrats_effect = yes
	}
}

country_event = {
	id = china_events.13
	title = "china_events.13.t"
	desc = "china_events.13.d"
	picture = MILITARY_CAMP_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		OR = {
			faction_in_power = es_knights
			AND = {
				has_dlc = "Mandate of Heaven"
				is_emperor_of_china = yes
			}
		}
		any_owned_province = {
			is_capital = no
			is_state = yes
		}
	}
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					is_state = yes
					is_capital = no
				}
				save_event_target_as = levy_province
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = "china_events.13.a"
		boost_temples_effect = yes
		event_target:levy_province = {
			add_base_manpower = 2
		}
	}
	
	option = {
		name = "china_events.13.b"
		boost_bureaucrats_effect = yes
		boost_eunuchs_effect = yes
		event_target:levy_province = {
			if = {
				limit = { has_dlc = "Mandate of Heaven" }
				if = {
					limit = { NOT = { prosperity = 100 } }
					add_prosperity = 50
				}
				add_base_tax = 1
			}
			if = {
				limit = { NOT = { has_dlc = "Mandate of Heaven" } }
				add_base_tax = 2
			}
		}
	}
}
