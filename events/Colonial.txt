### Events to give persecuted minorities into colonies ###

province_event = {
	id = colonial.5
	title = "EVTNAME1423"
	desc = "EVTDESC1423"
	picture = STREET_SPEECH_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		is_overseas = yes
		is_colony = yes
		colonysize = 600
		NOT = { has_province_modifier = religious_migration }
		owner = { 
			any_owned_province = {
				is_overseas = no
				religion_group = ROOT
				NOT = { religion = ROOT }
			}
		}
	}
	
	immediate = {
		hidden_effect = {
			owner = {
				random_owned_province = {
					limit = {
						is_overseas = no
						religion_group = ROOT
						NOT = { religion = ROOT }
					}
					save_event_target_as = heretics_from_here
				}
			}
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = "EVTOPTA1423"
		change_religion = event_target:heretics_from_here
		change_culture = owner
		add_colonysize = 400
		add_province_modifier = {
			name = "religious_migration"
			duration = 8760
		}
	}
	option = {
		name = "colonial.5.b"
		add_colonysize = -400
	}
}

# Fur Trade Dwindles
province_event = {
	id = colonial.13
	title = "colonial.EVTNAME13"
	desc = "colonial.EVTDESC13"
	picture = MERCHANTS_TALKING_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		trade_goods = fur
		OR = {
			NOT = { has_province_flag = received_trade_goods }
			had_province_flag = { flag = received_trade_goods days = 3650 }
		}
		NOT = { has_empty_adjacent_province = yes }
		NOT = {
			any_neighbor_province = {
				owner = {
					has_reform = steppe_horde
				}
				controller = {
					has_reform = steppe_horde
				}
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = "colonial.EVTOPTA13"
		change_trade_goods = naval_supplies
	}
	option = {
		name = "colonial.EVTOPTC13"
		change_trade_goods = grain
	}
}

# New World Coffee
province_event = {
	id = colonial.18
	title = "colonial.EVTNAME18"
	desc = "colonial.EVTDESC18"
	picture = TRADE_GOODS_PLANTATION_GOODS_IN_WAREHOUSE_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		OR = {
			NOT = { has_province_flag = received_trade_goods }
			had_province_flag = { flag = received_trade_goods days = 3650 }
		}
		OR = {
			trade_goods = grain
			trade_goods = fish
			trade_goods = wool
			trade_goods = naval_supplies
			trade_goods = salt
			trade_goods = fur
		}
		NOT = { has_empty_adjacent_province = yes }
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = "colonial.EVTOPTA18"
		ai_chance = { factor = 90 }
		change_trade_goods = coffee
	}
	option = {
		name = "colonial.EVTOPTC18"
		ai_chance = { factor = 10 }
	}
}
