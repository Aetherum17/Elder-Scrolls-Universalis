	valid_for_subject = no

	prerequisites = {
		NOT = { truce_with = FROM }
		NOT = { alliance_with = FROM }
		is_great_power = yes
		FROM = { is_great_power = yes }
		OR = {
		ai = no
		AND = {
		ai = yes
		FROM = { ai = no }
		}
		}
	}

	war_goal = ge_empire
}
